56 bool selectDecoder(const char* mime) {
57 bool result = false;
58 // do nothing if no change
59 if (StrView(mime).equals(actual_decoder.mime)) {
60 return true;
61 }
62 // close actual decoder
63 end();
64
65 // find the corresponding decoder
66 for (int j = 0; j < decoders.size(); j++) {
67 DecoderInfo info = decoders[j];
68 if (StrView(info.mime).equals(mime)) {
69 LOGI("New decoder found for %s (%s)", info.mime, mime);
70 actual_decoder = info;
71 // define output if it has not been defined
72 if (p_print!=nullptr
73 && actual_decoder.decoder->getOutput()==nullptr){
74 actual_decoder.decoder->setOutput(*p_print);
75 }
76 actual_decoder.decoder->begin();
77 result = true;
78 }
79 }
80 return result;
81 }

83 size_t write(const uint8_t* data, size_t len) override {
84 if (is_first) {

86 mime_detector.write((uint8_t*)data, len);
