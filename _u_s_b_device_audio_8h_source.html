648  virtual bool feature_unit_get_request(
649  uint8_t rhport, tusb_control_request_t const *p_request) {
650  // Page 91 in UAC2 specification
651  uint8_t channelNum = TU_U16_LOW(p_request->wValue);
652  uint8_t ctrlSel = TU_U16_HIGH(p_request->wValue);
653  // uint8_t itf = TU_U16_LOW(p_request->wIndex); // Since
654  // we have only one audio function implemented, we do not need the itf value
655  uint8_t entityID = TU_U16_HIGH(p_request->wIndex);
656  switch (ctrlSel) {
657  case AUDIO_FU_CTRL_MUTE: {
658  // Audio control mute cur parameter block consists of only one byte - we
659  // thus can send it right away There does not exist a range parameter
660  // block for mute
661  LOG_AUDIO_DEBUG(" Get Mute of channel: %u", channelNum);
662  bool current_mute = _mute[channelNum];
663  bool rc = tud_control_xfer(rhport, p_request, &current_mute, 1);
664  _mute[channelNum] = current_mute;
665  return rc;
666  }
667 
668  case AUDIO_FU_CTRL_VOLUME:
