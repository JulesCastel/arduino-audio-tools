209 void parsePacket() {
210 TRACEI();
211 uint8_t *packet = buffer.data();
212 int pid = ((packet[1] & 0x1F) << 8) | (packet[2] & 0xFF);
213 LOGI("PID: 0x%04X(%d)", pid, pid);
214 int payloadUnitStartIndicator = (packet[1] & 0x40) >> 6;
215 LOGI("Payload Unit Start Indicator: %d", payloadUnitStartIndicator);
216 int adaptionFieldControl = (packet[3] & 0x30) == 0x20;
217 LOGI("Adaption Field Control: %d", adaptionFieldControl);
218
219 bool has_payload = true;
220 if ((packet[3] & 0x10) >> 4 == 0) {
221 has_payload = false;
222 }
223
224 // Locate start of PSI section (skip adaptation field if present)
225 int payload_offset = 4;
226 if ((packet[3] & 0x20) >> 5 == 1) { // Adaptation field exists
227 payload_offset += packet[4] + 1;
228 }
229
230 // Adjust for the Pointer Field (only if PUSI is set)
231 if (payloadUnitStartIndicator) {
232 payload_offset += packet[payload_offset] + 1;
233 }
234
235 // if we are at the beginning we start with a pat
236 if (pid == 0 && payloadUnitStartIndicator) {
237 pids.clear();
238 }
239
240 int payloadStart = payload_offset;
241 int len = TS_PACKET_SIZE - payloadStart;
242
243 // PID 0 is for PAT
244 if (pid == 0) {
245 parsePAT(&packet[payloadStart], len);
246 } else if (pid == pmt_pid && packet[payloadStart] == 0x02) {
247 parsePMT(&packet[payloadStart], len);
248 } else if (pids.contains(pid)) {
249 parsePES(&packet[payloadStart], payloadUnitStartIndicator ? true : false,
250 len);
251 } else {
252 LOGE("-> Packet ignored for %d", pid);
253 }
254 }

256 void parsePAT(uint8_t *pat, int len) {
257 TRACEI();
258 assert(pat[0] == 0); // Program Association section
259 int startOfProgramNums = 8;
260 int lengthOfPATValue = 4;
261 int sectionLength = ((pat[1] & 0x0F) << 8) | (pat[2] & 0xFF);
262 LOGI("PAT Section Length: %d", sectionLength);
263 if (sectionLength >= len) {
264 LOGE("Unexpected PAT Section Length: %d", sectionLength);
265 sectionLength = len;
266 }
267 int indexOfPids = 0;
268 for (int i = startOfProgramNums; i <= sectionLength;
269 i += lengthOfPATValue) {
270 int program_number = ((pat[i] & 0xFF) << 8) | (pat[i + 1] & 0xFF);
271 int pid = ((pat[i + 2] & 0x1F) << 8) | (pat[i + 3] & 0xFF);
272 LOGI("Program Num: 0x%04X(%d) / PID: 0x%04X(%d) ", program_number,
273 program_number, pid, pid);
274
275 if (pmt_pid == 0xFFFF && pid >= 0x0020 && pid <= 0x1FFE) {
276 pmt_pid = pid;
277 }
278 }
279 LOGI("Using PMT PID: 0x%04X(%d)", pmt_pid, pmt_pid);

282 void parsePMT(uint8_t *pat, int len) {
283 TRACEI();
284 assert(pat[0] == 0x02); // Program Association section
285 int staticLengthOfPMT = 12;
286 int sectionLength = ((pat[1] & 0x0F) << 8) | (pat[2] & 0xFF);
287 LOGI("PMT Section Length: %d", sectionLength);
288 int programInfoLength = ((pat[10] & 0x0F) << 8) | (pat[11] & 0xFF);
289 LOGI("PMT Program Info Length: %d", programInfoLength);
290
291 int cursor = staticLengthOfPMT + programInfoLength;
292 while (cursor < sectionLength - 1) {
293 int streamType = pat[cursor] & 0xFF;
294 int elementaryPID =
295 ((pat[cursor + 1] & 0x1F) << 8) | (pat[cursor + 2] & 0xFF);
296 LOGI("Stream Type: 0x%02X(%d) Elementary PID: 0x%04X(%d)", streamType,
297 streamType, elementaryPID, elementaryPID);
298
299 if (isStreamTypeActive((MTSStreamType)streamType)) {
300 addPID(elementaryPID);
301 }
302
303 int esInfoLength =
304 ((pat[cursor + 3] & 0x0F) << 8) | (pat[cursor + 4] & 0xFF);
305 LOGI("ES Info Length: 0x%04X(%d)", esInfoLength, esInfoLength);
306 cursor += 5 + esInfoLength;
307 }
308 }

313 if (isNewPayload) {
314 uint8_t streamID = pat[3] & 0xFF;
