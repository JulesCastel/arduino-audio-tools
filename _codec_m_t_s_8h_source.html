260 void parsePacket() {
261 TRACEI();
262 uint8_t *packet = buffer.data();
263 int pid = ((packet[1] & 0x1F) << 8) | (packet[2] & 0xFF);
264 LOGI("PID: 0x%04X(%d)", pid, pid);
265 bool payloadUnitStartIndicator = false;
266 uint8_t adaptionField = (packet[3] & 0x30) >> 4;
267 int adaptationSize = 0;
268 int offset = 4; // Start after TS header (4 bytes)
269 bool to_process = true;
270
271 // Check for adaptation field
272 // 00 (0) → Invalid (should never happen).
273 // 01 (1) → Payload only (no adaptation field).
274 // 10 (2) → Adaptation field only (no payload).
275 // 11 (3) → Adaptation field + payload.
276 if (adaptionField == 0b11) { // Adaptation field exists
277 adaptationSize = packet[4] + 1;
278 offset += adaptationSize;
279 } else if (adaptionField == 0b10) { // Adopation field only
280 to_process = false;
281 }
282
283 // If PUSI is set, there's a pointer field (skip it)
284 if (packet[1] & 0x40) {
285 offset += packet[offset] + 1;
286 payloadUnitStartIndicator = true;
287 }
288
289 LOGI("Payload Unit Start Indicator (PUSI): %d", payloadUnitStartIndicator);
290 LOGI("Adaption Field Control: 0x%x / size: %d", adaptionField,
291 adaptationSize);
292
293 // if we are at the beginning we start with a pat
294 if (pid == 0 && payloadUnitStartIndicator) {
295 // pids.clear();
296 }
297
298 int payloadStart = offset;
299 int len = TS_PACKET_SIZE - payloadStart;
300
301 // PID 0 is for PAT
302 if (pid == 0) {
303 parsePAT(&packet[payloadStart], len);
304 } else if (pid == pmt_pid && packet[payloadStart] == 0x02) {
305 parsePMT(&packet[payloadStart], len);
306 } else if (!is_adts_missing && to_process && pids.contains(pid)) {
307 parsePES(&packet[payloadStart], len, payloadUnitStartIndicator);
308 } else {
309 LOGE("-> Packet ignored for PID 0x%x", pid);
310 }
311 }

313 void parsePAT(uint8_t *pat, int len) {
314 TRACEI();
315 assert(pat[0] == 0); // Program Association section
316 int startOfProgramNums = 8;
317 int lengthOfPATValue = 4;
318 int sectionLength = ((pat[1] & 0x0F) << 8) | (pat[2] & 0xFF);
319 LOGI("PAT Section Length: %d", sectionLength);
320 if (sectionLength >= len) {
321 LOGE("Unexpected PAT Section Length: %d", sectionLength);
322 sectionLength = len;
323 }
324 int indexOfPids = 0;
325 for (int i = startOfProgramNums; i <= sectionLength;
326 i += lengthOfPATValue) {
327 int program_number = ((pat[i] & 0xFF) << 8) | (pat[i + 1] & 0xFF);
328 int pid = ((pat[i + 2] & 0x1F) << 8) | (pat[i + 3] & 0xFF);
329 LOGI("Program Num: 0x%04X(%d) / PID: 0x%04X(%d) ", program_number,
330 program_number, pid, pid);
331
332 if (pmt_pid == 0xFFFF && pid >= 0x0020 && pid <= 0x1FFE) {
333 pmt_pid = pid;
334 }
335 }
336 LOGI("Using PMT PID: 0x%04X(%d)", pmt_pid, pmt_pid);
337 }

339 void parsePMT(uint8_t *pmt, int len) {
340 TRACEI();
341 assert(pmt[0] == 0x02); // Program Association section
342 int staticLengthOfPMT = 12;
343 int sectionLength = ((pmt[1] & 0x0F) << 8) | (pmt[2] & 0xFF);
344 LOGI("- PMT Section Length: %d", sectionLength);
345 int programInfoLength = ((pmt[10] & 0x0F) << 8) | (pmt[11] & 0xFF);
346 LOGI("- PMT Program Info Length: %d", programInfoLength);
347
348 int cursor = staticLengthOfPMT + programInfoLength;
349 while (cursor < sectionLength - 1) {
350 MTSStreamType streamType = static_cast<MTSStreamType>(pmt[cursor] & 0xFF);
351 int elementaryPID =
352 ((pmt[cursor + 1] & 0x1F) << 8) | (pmt[cursor + 2] & 0xFF);
353 LOGI("-- Stream Type: 0x%02X(%d) [%s] for Elementary PID: 0x%04X(%d)",
354 streamType, streamType, toStr(streamType), elementaryPID,
355 elementaryPID);
356
357 if (isStreamTypeActive(streamType)) {
358 selected_stream_type = streamType;
359 addPID(elementaryPID);
360 }
361
362 int esInfoLength =
363 ((pmt[cursor + 3] & 0x0F) << 8) | (pmt[cursor + 4] & 0xFF);
364 LOGI("-- ES Info Length: 0x%04X(%d)", esInfoLength, esInfoLength);
365 cursor += 5 + esInfoLength;
366 }
367 }

369 void parsePES(uint8_t *pes, int len, const bool isNewPayload) {
370 LOGI("parsePES: %d", len);
371 ++pes_count;
372 uint8_t *data = nullptr;
373 int dataSize = 0;
374
375 if (isNewPayload) {
376 assert(len >= 6);
377 if (!isPESStartCodeValid(pes)) {
378 LOGI("PES header not aligned correctly - adjusting");
379 pes--;
380 }
381 // PES header is not alligned correctly
382 if (!isPESStartCodeValid(pes)) {
383 LOGE("PES header not aligned correctly");
384 return;
385 }
386
387 int pesPacketLength =
388 (static_cast<int>(pes[4]) << 8) | static_cast<int>(pes[5]);
389
390 // PES Header size is at least 6 bytes, but can be larger with optional
391 // fields
392 int pesHeaderSize = 6;
393 if ((pes[6] & 0xC0) != 0) { // Check for PTS/DTS flags
394 pesHeaderSize += 3 + ((pes[7] & 0xC0) == 0xC0 ? 5 : 0);
395 pesHeaderSize += pes[8]; // PES header stuffing size
396 }
397 LOGI("- PES Header Size: %d", pesHeaderSize);
398 data = pes + pesHeaderSize;
399 dataSize = len - pesHeaderSize;
400
401 assert(pesHeaderSize < len);
402 assert(dataSize > 0);
403
405 if (pes_count == 1 && selected_stream_type == MTSStreamType::AUDIO_AAC) {
406 is_adts_missing = findSyncWord(data, dataSize) == -1;
407 }
408
409 open_pes_data_size = pesPacketLength;
410
411 } else {
412 data = pes;
413 dataSize = len;
414 }
415 open_pes_data_size -= dataSize;
416
417 if (open_pes_data_size < 0){
418 return;
419 }
420
421 LOGI("- writing %d bytes (open: %d)", dataSize, open_pes_data_size);
422 if (p_print) {
423 size_t result = writeData<uint8_t>(p_print, data, dataSize);
424 assert(result==dataSize);
425 }
426 if (p_dec) {
427 size_t result = writeDataT<uint8_t, AudioDecoder>(p_dec, data, dataSize);
428 assert(result==dataSize);
429 }
430 }
