200 static bool parseData(OSCData &osc, void *ref) {
201 uint64_t time = osc.readTime();
202 uint64_t seq = osc.readTime();
203 OSCBinaryData data = osc.readData();
204 OSCContainerDecoder *self = static_cast<OSCContainerDecoder *>(ref);
205 // store the actual sequence number
206 self->seq_no = seq;
207 // call write callbak if defined
208 if (self->write_callback != nullptr) {
209 bool ok = self->write_callback(time, seq, data.data, data.len, ref);
210 if (!ok) return true;
211 }
212 // output to decoder
213 if (self->p_codec != nullptr) {
214 self->p_codec->write(data.data, data.len);
215 }
216 return true;
217 }

219 static bool parseInfo(OSCData &osc, void *ref) {
220 AudioInfo info;
221 info.sample_rate = osc.readInt32();
222 info.channels = osc.readInt32();
223 info.bits_per_sample = osc.readInt32();
224 const char *mime = osc.readString();
225
226 OSCContainerDecoder *self = static_cast<OSCContainerDecoder *>(ref);
227 if (self != nullptr) {
228 self->setAudioInfo(info);
229 self->mime_str = mime;
230 LOGI("mime: %s", mime);
231 // select the right decoder based on the mime type
232 if (self->p_multi_decoder != nullptr)
233 self->p_multi_decoder->selectDecoder(mime);
234 }
235
236 return true;
237 }
