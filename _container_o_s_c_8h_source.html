223 static bool parseData(OSCData &osc, void *ref) {
224 uint64_t time = osc.readTime();
225 uint64_t seq = osc.readTime();
226 OSCBinaryData data = osc.readData();
227 OSCContainerDecoder *self = static_cast<OSCContainerDecoder *>(ref);
228 // Check for missing sequence numbers
229 if (self->seq_no + 1 != seq) {
230 self->missing_data_callback(self->seq_no + 1, seq - 1, self->ref);
231 }
232 // store the actual sequence number
233 self->seq_no = seq;
234 // call write callbak if defined
235 if (self->write_callback != nullptr) {
236 bool ok = self->write_callback(time, seq, data.data, data.len, ref);
237 if (!ok) return true;
238 }
239 // output to decoder
240 if (self->p_codec != nullptr) {
241 self->p_codec->write(data.data, data.len);
242 }
243
244 return true;
245 }
246
247 static bool parseInfo(OSCData &osc, void *ref) {
248 AudioInfo info;
249 info.sample_rate = osc.readInt32();
250 info.channels = osc.readInt32();
251 info.bits_per_sample = osc.readInt32();
252 const char *mime = osc.readString();
253
254 OSCContainerDecoder *self = static_cast<OSCContainerDecoder *>(ref);
255 if (self != nullptr) {
256 self->setAudioInfo(info);
257 self->mime_str = mime;
258 LOGI("mime: %s", mime);
259 // select the right decoder based on the mime type
260 if (self->p_multi_decoder != nullptr)
261 self->p_multi_decoder->selectDecoder(mime);
262 }
263
264 return true;
265 }
