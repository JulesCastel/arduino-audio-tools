116  bool findTag(const uint8_t *data, size_t len, int &pos_tag, int &meta_len) {
117  MetaType tag_type;

119  switch (tag_type) {

149  bool find(const char *str, size_t len, int &pos, MetaType &type) {
150  if (str == nullptr || len <= 0) return false;
151  for (size_t j = 0; j <= len - 3; j++) {
152  if (str[j] == 'T' && str[j + 1] == 'A' && str[j + 2] == 'G') {
153  type = str[j + 3] == '+' ? TAG_PLUS : TAG;
154  pos = j;
155  return true;
156  } else if (str[j] == 'I' && str[j + 1] == 'D' && str[j + 2] == '3') {
157  type = ID3;
158  pos = j;
159  return true;
160  }
161  }
162  return false;
163  }
