141 void effect(AudioFFTBase &fft) {
142 TRACED();
143 FFTBin bin;
144 for (int n = 0; n < fft.size(); n++) {
145 float amplitude = fft.magnitude(n);
146 float phase = rand() / (float)RAND_MAX * 2.f * M_PI;
147
148 // update new bin value
149 bin.real = cosf(phase) * amplitude;
150 bin.img = sinf(phase) * amplitude;
151 fft.setBin(n, bin);

173 friend FFTEffect;
174
175 public:
176 FFTPitchShift(AudioStream &out) : FFTEffect(out) {
177 addNotifyAudioChange(out);

184 FFTPitchShiftConfig defaultConfig() {
185 FFTPitchShiftConfig result;
186 result.shift = shift;
187 return result;
188 }
189
190 bool begin(FFTPitchShiftConfig psConfig) {
191 setShift(psConfig.shift);
192 return FFTEffect::begin(psConfig);
193 }
194
196 void setShift(int bins) { shift = bins; }
197
198 protected:
199 int shift = 1;
200
202 void effect(AudioFFTBase &fft) {
203 TRACED();
204 FFTBin bin;
205 int max = fft.size();
206
207 if (shift < 0) {
208 // copy bins: left shift
209 for (int n = -shift; n < max; n++) {
210 int to_bin = n + shift;
211 assert(to_bin >= 0);
212 assert(to_bin < max);
213 fft.getBin(n, bin);
214 fft.setBin(to_bin, bin);
215 }
216 // clear tail
217 bin.clear();
218 for (int n = max + shift; n < max; n++) {
219 fft.setBin(n, bin);
220 }
221 } else if (shift > 0) {
222 // copy bins: right shift
223 for (int n = max - shift; n <= 0; n--) {
224 int to_bin = n + shift;
225 assert(to_bin >= 0);
226 assert(to_bin < max);
227 fft.getBin(n, bin);
228 fft.setBin(to_bin, bin);
229 }
230 // clear head
231 bin.clear();
232 for (int n = 0; n < shift; n++) {
233 fft.setBin(n, bin);
234 }
235 }
236 }
237};
