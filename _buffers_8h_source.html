440 int readArray(T data[], int count) override {

443
444 OffsetInfo offset = getOffset(read_pos, read_count);

545 struct OffsetInfo {
546 int pos = 0; // start pos
547 int len = 0; // length of first part
548 int len1 = 0; // length of second part on overflow
549 };
550
552 OffsetInfo getOffset(int pos, int len) {
553 OffsetInfo result;
554 result.pos = pos;
555 int overflow = (pos + len) - max_size;
556 if (overflow <= 0) {
557 // we can write the complete data
558 result.len = len;
559 result.len1 = 0;
560 } else {
561 // we need to split the data
562 result.len = len - overflow;
563 result.len1 = overflow;
564 }
565 return result;
566 }
567
569 bool file_seek(int pos) {
570 int file_pos = pos * sizeof(T);
571 if (p_file->position() != file_pos) {
572 LOGD("file_seek: %d", pos);
573 if (!p_file->seek(file_pos)) {
574 LOGE("seek %d", file_pos);
575 return false;
576 }
577 }
578 return true;
579 }
580
582 int file_write(const T *data, int count) {
583 LOGD("file_write: %d", count);
584 if (p_file == nullptr) return 0;
585 int to_write = sizeof(T) * count;
586 int bytes_written = p_file->write((const uint8_t *)data, to_write);
587 p_file->flush();
588 int elements_written = bytes_written / sizeof(T);
589 if (bytes_written != to_write) {
590 LOGE("write: %d -> %d", to_write, bytes_written);
591 }
592 return elements_written;
593 }
594
596 int file_read(T *result, int count) {
597 LOGD("file_read: %d", count);
598 int read_bytes = count * sizeof(T);
599 int result_bytes = p_file->read((uint8_t *)result, read_bytes);
600 int result_count = result_bytes / sizeof(T);
601 if (result_count != count) {
602 LOGE("readBytes: %d -> %d", read_bytes, result_bytes);
603 }
604 return result_count;
605 }

615template <typename T>
616class NBuffer : public BaseBuffer<T> {
617 public:
618 NBuffer(int size, int count) { resize(size, count); }
619
620 virtual ~NBuffer() { freeMemory(); }
621
622 // reads an entry from the buffer
623 bool read(T &result) override {
624 if (available() == 0) return false;
625 return actual_read_buffer->read(result);
626 }
627
628 // peeks the actual entry from the buffer
629 bool peek(T &result) override {
630 if (available() == 0) return false;
631 return actual_read_buffer->peek(result);
632 }
633
634 // checks if the buffer is full
635 bool isFull() { return availableForWrite() == 0; }
636
637 // write add an entry to the buffer
638 bool write(T data) {
639 bool result = false;
640 if (actual_write_buffer == nullptr) {
641 actual_write_buffer = getNextAvailableBuffer();
642 }
643 if (actual_write_buffer != nullptr) {
644 result = actual_write_buffer->write(data);
645 // if buffer is full move to next available
646 if (actual_write_buffer->isFull()) {
647 addFilledBuffer(actual_write_buffer);
648 actual_write_buffer = getNextAvailableBuffer();
649 }
650 } else {
651 // Logger.debug("actual_write_buffer is full");
652 }
653
654 if (start_time == 0l) {
655 start_time = millis();
656 }
657 if (result) sample_count++;
658
659 return result;
660 }
661
662 // determines the available entries for the current read buffer
663 int available() {
664 if (actual_read_buffer == nullptr) {
665 actual_read_buffer = getNextFilledBuffer();
666 }
667 if (actual_read_buffer == nullptr) {
668 return 0;
669 }
670 int result = actual_read_buffer->available();
671 if (result == 0) {
672 // make current read buffer available again
673 resetCurrent();
674 result =
675 (actual_read_buffer == nullptr) ? 0 : actual_read_buffer->available();
676 }
677 return result;
678 }
679
680 // deterMINes the available entries for the write buffer
681 int availableForWrite() {
682 if (actual_write_buffer == nullptr) {
683 actual_write_buffer = getNextAvailableBuffer();
684 }
685 // if we used up all buffers - there is nothing available any more
686 if (actual_write_buffer == nullptr) {
687 return 0;
688 }
689 // check on actual buffer
690 if (actual_write_buffer->isFull()) {
691 // if buffer is full we move it to filled buffers ang get the next
692 // available
693 addFilledBuffer(actual_write_buffer);
694 actual_write_buffer = getNextAvailableBuffer();
695 }
696 return actual_write_buffer->availableForWrite();
697 }
698
699 // resets all buffers
700 void reset() {
701 TRACED();
702 while (actual_read_buffer != nullptr) {
703 actual_read_buffer->reset();
704 addAvailableBuffer(actual_read_buffer);
705 // get next read buffer
706 actual_read_buffer = getNextFilledBuffer();
707 }
708 }
709
710 // provides the actual sample rate
711 unsigned long sampleRate() {
712 unsigned long run_time = (millis() - start_time);
713 return run_time == 0 ? 0 : sample_count * 1000 / run_time;
714 }
715
716 // returns the address of the start of the phsical read buffer
717 T *address() {
718 return actual_read_buffer == nullptr ? nullptr
719 : actual_read_buffer->address();
720 }
721
722 // Alternative interface using address: the current buffer has been filled
723 BaseBuffer<T> &writeEnd() {
724 if (actual_write_buffer != nullptr) {
725 actual_write_buffer->setWritePos(buffer_size);
726 addFilledBuffer(actual_write_buffer);
727 }
728 actual_write_buffer = getNextAvailableBuffer();
729 return *actual_write_buffer;
730 }
731
732 // Alternative interface using address: marks actual buffer as processed and
733 // provides access to next read buffer
734 BaseBuffer<T> &readEnd() {
735 // make current read buffer available again
736 resetCurrent();
737 return *actual_read_buffer;
738 }
739
740 virtual int bufferCountFilled() { return filled_buffers.size(); }
741
742 virtual int bufferCountEmpty() { return available_buffers.size(); }
743
744 virtual void resize(int size, int count) {
745 if (buffer_size == size && buffer_count == count) return;
746 freeMemory();
747 filled_buffers.resize(count);
748 available_buffers.resize(count);
749 // filled_buffers.clear();
750 // available_buffers.clear();
751
752 buffer_count = count;
753 buffer_size = size;
754 for (int j = 0; j < count; j++) {
755 BaseBuffer<T> *buffer = new SingleBuffer<T>(size);
756 LOGD("new buffer %p", buffer);
757 available_buffers.enqueue(buffer);
758 }
759 }
760
761 size_t size() { return buffer_size * buffer_count; }
762
763 protected:
764 int buffer_size = 0;
765 uint16_t buffer_count = 0;
766 BaseBuffer<T> *actual_read_buffer = nullptr;
767 BaseBuffer<T> *actual_write_buffer = nullptr;
768 QueueFromVector<BaseBuffer<T> *> available_buffers{0, nullptr};
769 QueueFromVector<BaseBuffer<T> *> filled_buffers{0, nullptr};
770 // Queue<BaseBuffer<T> *> available_buffers;
771 // Queue<BaseBuffer<T> *> filled_buffers;
772 unsigned long start_time = 0;
773 unsigned long sample_count = 0;
774
775 // empty constructor only allowed by subclass
776 NBuffer() = default;
777
778 void freeMemory() {
779 if (actual_write_buffer) {
780 LOGD("deleting %p", actual_write_buffer);
781 delete actual_write_buffer;
782 actual_write_buffer = nullptr;
783 }
784 if (actual_read_buffer) {
785 LOGD("deleting %p", actual_read_buffer);
786 delete actual_read_buffer;
787 actual_read_buffer = nullptr;
788 }
789
790 BaseBuffer<T> *ptr = getNextAvailableBuffer();
791 while (ptr != nullptr) {
792 LOGD("deleting %p", ptr);
793 delete ptr;
794 ptr = getNextAvailableBuffer();
795 }
796
797 ptr = getNextFilledBuffer();
798 while (ptr != nullptr) {
799 LOGD("deleting %p", ptr);
800 delete ptr;
801 ptr = getNextFilledBuffer();
802 }
803 }
804
805 void resetCurrent() {
806 if (actual_read_buffer != nullptr) {
807 actual_read_buffer->reset();
808 addAvailableBuffer(actual_read_buffer);
809 }
810 // get next read buffer
811 actual_read_buffer = getNextFilledBuffer();
812 }
813
814 virtual BaseBuffer<T> *getNextAvailableBuffer() {
815 if (available_buffers.empty()) return nullptr;
816 BaseBuffer<T> *result = nullptr;
817 available_buffers.dequeue(result);
818 return result;
819 }
820
821 virtual bool addAvailableBuffer(BaseBuffer<T> *buffer) {
822 return available_buffers.enqueue(buffer);
823 }
824
825 virtual BaseBuffer<T> *getNextFilledBuffer() {
826 if (filled_buffers.empty()) return nullptr;
827 BaseBuffer<T> *result = nullptr;
828 filled_buffers.dequeue(result);
829 return result;
830 }
831
832 virtual bool addFilledBuffer(BaseBuffer<T> *buffer) {
833 return filled_buffers.enqueue(buffer);
834 }

987 int16_t *getValues(size_t offset, size_t length) {
988 LOGD("getValues(%d,%d) - max %d", offset, length, array.size());
989 if (offset == 0) {
990 // we restart at the beginning
991 last_end = 0;
992 actual_end = length;
993 } else {
994 // if first position is at end we do not want to read the full buffer
995 last_end = actual_end >= 0 ? actual_end : offset;
996 // increase actual end if bigger then old
997 actual_end = offset + length > actual_end ? offset + length : actual_end;
998 }
999 int size = actual_end - last_end;
1000 if (size > 0) {
1001 LOGD("readBytes(%d,%d)", last_end, size);
1002 assert(last_end + size <= array.size());
1003 p_stream->readBytes((uint8_t *)(&array[last_end]), size * 2);
1004 }
1005 assert(offset < actual_end);
1006 return &array[offset];
1007 }
1008
1009 protected:
1010 int actual_end = -1;
1011 int last_end = 0;
1012 Vector<T> array;
1013 Stream *p_stream = nullptr;
1014};
