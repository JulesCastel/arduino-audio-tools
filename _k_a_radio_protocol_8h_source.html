146 bool processCommand(const char* input, Print& result) {
147 if (p_player == nullptr) {
148 LOGE("player not set");
149 return false;
150 }
151
152 Str name;
153 Str arg;
154 StrView line = input;
155
156 int start = line.indexOf('?');
157 if (start < 0) start = 0;
158 int endPos = line.length();
159 bool rc = true;
160 while (start >= 0 && start < endPos) {
161 int eqPos = line.indexOf('=', start);
162 int toPos = getEndPos(line, start + 1);
163 if (eqPos >= 0) {
164 name.substring(line, start + 1, eqPos);
165 arg.substring(line, eqPos + 1, toPos);
166 } else {
167 name.substring(line, start + 1, toPos);
168 arg = "";
169 }
170 LOGD("start=%d, eqPos=%d, toPos=%d", start, eqPos, toPos);
171 // remove leading and trailing spaces
172 name.trim();
173 arg.trim();
174 // execute the command
175 rc = processCommand(name, arg, result);
176 // calculate new positions
177 start = toPos;
178 toPos = getEndPos(line, start + 1);
179 }
180 return rc;
181 }

184 bool processCommand(Str& name, Str& arg, Print& result) {
185 LOGI("command: %s (%s)", name.c_str(), arg.c_str());
186 assert(p_player != nullptr);
187
188 for (Action& act : actions) {
189 if (name.equals(act.cmd)) {
190 return act.callback(*p_player, name, arg, result, this);
191 }
192 }
193
194 LOGE("Invalid command:", name.c_str());
195 return false;
196 }
